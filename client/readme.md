# browser-sync


```sh

# step 1
$ npm install -g browser-sync
# OR
$ npm i -g browser-sync


# step 2
$ cd build

# step 3
$ browser-sync start --server --files "./*.html"


$ browser-sync start --server --files "./*.*"
$ browser-sync start --server --files "./*.js"
$ browser-sync start --server --files "./*.css"

```
